"use client"
import { useState } from 'react';
// Import Css
import lineasProductosStyles from '../../styles/homeLineasProductosCompt.module.css'

// import Component
import BotonLineaProdCompt from './botonLineaProdCompt'; 

export default function HomeLineasProductosCompt() {
    const [index, setIndex] = useState(0)

    
    const [touchStartX, setTouchStartX] = useState(0);
    const [touchEndX, setTouchEndX] = useState(0);

    const handleTouchStart = (e: React.TouchEvent) => {
        setTouchStartX(e.targetTouches[0].clientX);
    };

    const handleTouchMove = (e: React.TouchEvent) => {
        setTouchEndX(e.targetTouches[0].clientX);
    };

    const handleTouchEnd = () => {
        if (touchStartX - touchEndX > 50) {
        // deslizó a la izquierda → siguiente slide
        setIndex(Math.min(index + 1, 1));
        }
        
        if (touchStartX - touchEndX < -50) {
        // deslizó a la derecha → slide anterior
        setIndex(Math.max(index - 1, 0));
        }
    };

    
    return(

        <div className={lineasProductosStyles.contentLProdCompt}>

            <p>Líneas de Productos</p>


            <div className={lineasProductosStyles.contentSlideLP}>

                <button 

                    className={`${lineasProductosStyles.botonSlideMove} ${lineasProductosStyles.botonSlideIzq}`} 
                    onClick={() => setIndex(0)}>

                    <span className="material-symbols-outlined">arrow_back_ios</span>

                </button>


                 <div
                    className={lineasProductosStyles.sliderTrack}
                    style={{ transform: `translateX(-${index * 100}%)` }}
                    onTouchStart={handleTouchStart}
                    onTouchMove={handleTouchMove}
                    onTouchEnd={handleTouchEnd}
                    >

                    <div className={`${lineasProductosStyles.boxSlide} ${lineasProductosStyles.boxSlideA}`}>

                            {/* Botón No. 1 */}
                            <BotonLineaProdCompt 
                                bkUrlImagen = "/bkBotonlineaProd/bkLineaProd01.jpg"
                                hrefLinkToPage = "/tienda-linea"
                                nameLinkBoton = "Ver Productos"
                                titleLineaProd= {<>Equipos de <br /> Laboratorio</>}
                                filtroProd= {1}
                            />

                            {/* Botón No. 2 */}
                            <BotonLineaProdCompt 
                                bkUrlImagen = "/bkBotonlineaProd/bkLineaProd02.jpg"
                                hrefLinkToPage = "/tienda-linea"
                                nameLinkBoton = "Ver Productos"
                                titleLineaProd= {<>Instrumentos de <br /> Medición</>}
                                filtroProd={2}
                            />

                            {/* Botón No. 3 */}
                            <BotonLineaProdCompt 
                                bkUrlImagen = "/bkBotonlineaProd/bkLineaProd03.jpg"
                                hrefLinkToPage = "/tienda-linea"
                                nameLinkBoton = "Ver Productos"
                                titleLineaProd= {<>Consumibles de <br /> Precisión</>}
                                filtroProd={3}
                            />

                    </div>

                    <div className={`${lineasProductosStyles.boxSlide} ${lineasProductosStyles.boxSlideB}`}>


                            {/* Botón No. 4 */}
                            <BotonLineaProdCompt 
                                bkUrlImagen = "/bkBotonlineaProd/bkLineaProd04.jpg"
                                hrefLinkToPage = "/tienda-linea"
                                nameLinkBoton = "Ver Productos"
                                titleLineaProd= {<>Bioseguridad y <br /> Cadena de Frio</>}
                                filtroProd={4}
                            />

                            {/* Botón No. 5 */}
                            <BotonLineaProdCompt 
                                bkUrlImagen = "/bkBotonlineaProd/bkLineaProd05.jpg"
                                hrefLinkToPage = "/tienda-linea"
                                nameLinkBoton = "Ver Productos"
                                titleLineaProd= {<>Metrología y <br /> Control Industrial</>}
                                filtroProd={5}
                            />

                            {/* Botón No. 6 */}
                            <BotonLineaProdCompt 
                                bkUrlImagen = "/bkBotonlineaProd/bkLineaProd06.jpg"
                                hrefLinkToPage = "/tienda-linea"
                                nameLinkBoton = "Ver Productos"
                                titleLineaProd= {<>Modelos Anatómicos y <br /> Material Didáctico</>}
                                filtroProd={6}
                            />

                    </div>

                </div>

                             <button 
                                className={`${lineasProductosStyles.botonSlideMove} ${lineasProductosStyles.botonSlideDer}`} 
                                onClick={() => setIndex(1)}>
                                
                                <span className="material-symbols-outlined">arrow_forward_ios</span>
                                
                            </button>


            </div>

               

        </div>

    ); 

    
}
